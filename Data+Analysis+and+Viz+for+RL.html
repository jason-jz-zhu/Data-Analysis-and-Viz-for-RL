2.4 Heatmap¶

Heatmap¶

Based on the heatmap, we can find that rl has biggest count for highest level socre (>=190)¶

In [15]:
def score_category(score):
    if score < -150:
        return '< -150'
    elif -150 <= score < -100:
        return '-150 ~ -100'
    elif -100 <= score < -50:
        return '-100 ~ -50'
    elif -50 <= score < 0:
        return '-50 ~ 0'
    elif 0 <= score < 50:
        return '0 ~ 50'
    elif 50 <= score < 100:
        return '50 ~ 100'
    elif 100 <= score < 150:
        return '100 ~ 150'
    elif 150 <= score < 190:
        return '150 ~ 190'
    elif score >= 190:
        return '>= 190'
